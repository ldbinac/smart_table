# SmartTable 技术架构设计文档

## 一、架构总览

### 1.1 架构设计理念

SmartTable 采用**纯前端分层架构**，所有数据持久化在浏览器本地，通过 IndexedDB 实现类数据库的存储能力。

```
┌─────────────────────────────────────────────────────────────────┐
│                         表现层 (Presentation)                    │
│  ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐   │
│  │ 表格视图 │ │ 看板视图 │ │ 日历视图 │ │ 甘特视图 │ │ 仪表盘  │   │
│  └─────────┘ └─────────┘ └─────────┘ └─────────┘ └─────────┘   │

│  └─────────┘ └─────────┘ └─────────┘ └─────────┘ └─────────┘   │
```

```typescript
// db/services/baseService.ts
import { db } from '../schema';

export class BaseService {
  async createBase(data: Omit<Base, 'id' | 'createdAt' | 'updatedAt'>): Promise<Base> {
    const base: Base = {
      ...data,
      id: generateId(),
      createdAt: Date.now(),
      updatedAt: Date.now()
    };
    await db.bases.add(base);
    return base;
  }

  async getBase(id: string): Promise<Base | undefined> {
    return db.bases.get(id);
  }

  async getAllBases(): Promise<Base[]> {
    return db.bases.orderBy('updatedAt').reverse().toArray();
  }

  async updateBase(id: string, changes: Partial<Base>): Promise<void> {
    await db.bases.update(id, {
      ...changes,
      updatedAt: Date.now()
    });
  }

  async deleteBase(id: string): Promise<void> {
    await db.transaction('rw', 
      [db.bases, db.tables, db.fields, db.records, db.views, db.dashboards], 
      async () => {
        // 级联删除
        const tables = await db.tables.where('baseId').equals(id).toArray();
        for (const table of tables) {
          await this.deleteTableData(table.id);
        }
        await db.dashboards.where('baseId').equals(id).delete();
        await db.bases.delete(id);
      }
    );
  }

  private async deleteTableData(tableId: string): Promise<void> {
    await db.fields.where('tableId').equals(tableId).delete();
    await db.records.where('tableId').equals(tableId).delete();
    await db.views.where('tableId').equals(tableId).delete();
  }
}

export const baseService = new BaseService();
```

```typescript
// stores/baseStore.ts
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import { baseService } from '@/db/services/baseService';
import type { Base, Table, Field, Record, View } from '@/db/schema';

export const useBaseStore = defineStore('base', () => {
  // State
  const currentBase = ref<Base | null>(null);
  const tables = ref<Table[]>([]);
  const currentTable = ref<Table | null>(null);
  const fields = ref<Field[]>([]);
  const records = ref<Record[]>([]);
  const views = ref<View[]>([]);
  const currentView = ref<View | null>(null);
  const loading = ref(false);

  // Getters
  const primaryField = computed(() => 
    fields.value.find(f => f.isPrimary)
  );

  const visibleFields = computed(() => {
    if (!currentView.value) return fields.value;
    return fields.value.filter(f => 
      !currentView.value!.hiddenFields.includes(f.id)
    );
  });

  const filteredRecords = computed(() => {
    let result = [...records.value];
    
    // 应用筛选
    if (currentView.value?.filters.length) {
      result = applyFilters(result, currentView.value.filters, fields.value);
    }
    
    // 应用排序
    if (currentView.value?.sorts.length) {
      result = applySorts(result, currentView.value.sorts);
    }
    
    return result;
  });

  // Actions
  async function loadBase(baseId: string) {
    loading.value = true;
    try {
      currentBase.value = await baseService.getBase(baseId);
      if (currentBase.value) {
        tables.value = await tableService.getTablesByBase(baseId);
      }
    } finally {
      loading.value = false;
    }
  }

  async function loadTable(tableId: string) {
    loading.value = true;
    try {
      currentTable.value = await tableService.getTable(tableId);
      if (currentTable.value) {
        [fields.value, records.value, views.value] = await Promise.all([
          fieldService.getFieldsByTable(tableId),
          recordService.getRecordsByTable(tableId),
          viewService.getViewsByTable(tableId)
        ]);
        
        // 设置默认视图
        currentView.value = views.value.find(v => v.isDefault) || views.value[0] || null;
      }
    } finally {
      loading.value = false;
    }
  }

  async function createRecord(values: Record<string, any>) {
    if (!currentTable.value) return;
    
    const record = await recordService.createRecord({
      tableId: currentTable.value.id,
      values
    });
    
    records.value.push(record);
    return record;
  }

  async function updateRecord(recordId: string, values: Record<string, any>) {
    const record = records.value.find(r => r.id === recordId);
    if (!record) return;

    const oldValues = { ...record.values };
    const newValues = { ...oldValues, ...values };
    
    await recordService.updateRecord(recordId, newValues);
    
    record.values = newValues;
    record.updatedAt = Date.now();

    // 记录历史
    await historyService.addHistory({
      baseId: currentBase.value!.id,
      tableId: currentTable.value!.id,
      recordId,
      action: 'update',
      entityType: 'record',
      oldValue: oldValues,
      newValue: newValues
    });
  }

  return {
    currentBase,
    tables,
    currentTable,
    fields,
    records,
    views,
    currentView,
    loading,
    primaryField,
    visibleFields,
    filteredRecords,
    loadBase,
    loadTable,
    createRecord,
    updateRecord
  };
});
```

```typescript
  // 日历视图配置

    dateFieldId: viewConfig.value.dateFieldId,
    endDateFieldId: viewConfig.value.endDateFieldId,
    defaultView: viewConfig.value.defaultView || 'month',

  const groupedRecords = computed(() => {
    const groupFieldId = kanbanViewConfig.value.groupFieldId;
    if (!groupFieldId) return { '未分组': baseStore.filteredRecords };

    const groups: Record<string, Record[]> = {};
    const field = baseStore.fields.find(f => f.id === groupFieldId);
    
    // 初始化分组
    if (field?.type === FieldType.SINGLE_SELECT) {
      field.options?.options?.forEach(opt => {
        groups[opt.name] = [];
      });
    }
    groups['未分组'] = [];

    // 分组数据
    baseStore.filteredRecords.forEach(record => {
      const value = record.values[groupFieldId];
      const groupKey = value?.toString() || '未分组';
      if (!groups[groupKey]) groups[groupKey] = [];
      groups[groupKey].push(record);
    });

    return groups;
  });

  return {
    tableViewConfig,
    kanbanViewConfig,
    calendarViewConfig,
    groupedRecords,
  };
```

### 2.5 公式引擎 (Formula Engine)

#### 2.5.1 公式解析与计算

```typescript
// utils/formula/engine.ts
import { create, all } from 'mathjs';

const math = create(all, {
  number: 'number',
  precision: 14
});

export class FormulaEngine {
  private fields: Map<string, Field>;
  private records: Map<string, Record>;

  constructor(fields: Field[], records: Record[]) {
    this.fields = new Map(fields.map(f => [f.id, f]));
    this.records = new Map(records.map(r => [r.id, r]));
  }

  // 解析公式中的字段引用
  parseFieldRefs(formula: string): string[] {
    const refs: string[] = [];
    const regex = /\{([^}]+)\}/g;
    let match;
    while ((match = regex.exec(formula)) !== null) {
      const fieldName = match[1];
      const field = Array.from(this.fields.values()).find(f => f.name === fieldName);
      if (field) {
        refs.push(field.id);
      }
    }
    return refs;
  }

  // 计算单条记录的公式值
  calculate(record: Record, formula: string): any {
    try {
      // 替换字段引用为实际值
      let expression = formula;
      const fieldRefs = this.parseFieldRefs(formula);
      
      for (const fieldId of fieldRefs) {
        const field = this.fields.get(fieldId);
        if (!field) continue;
        
        const value = record.values[fieldId];
        const jsValue = this.toJsValue(value, field.type);
        
        expression = expression.replace(
          new RegExp(`\\{${field.name}\\}`, 'g'),
          jsValue
        );
      }

      // 执行计算
      const result = math.evaluate(expression);
      return this.fromJsValue(result);
    } catch (error) {
      console.error('Formula calculation error:', error);
      return '#ERROR';
    }
  }

  // 转换为JavaScript可用值
  private toJsValue(value: CellValue, fieldType: FieldType): string {
    if (value === null || value === undefined) return '0';
    
    switch (fieldType) {
      case FieldType.NUMBER:
        return String(value);
      case FieldType.TEXT:
        return `"${String(value).replace(/"/g, '\\"')}"`;
      case FieldType.CHECKBOX:
        return value ? '1' : '0';
      case FieldType.DATE:
        return String(new Date(value as string).getTime());
      default:
        return `"${String(value)}"`;
    }
  }

  // 转换回存储值
  private fromJsValue(value: any): CellValue {
    if (typeof value === 'number') {
      return Math.round(value * 10000) / 10000; // 保留4位小数
    }
    return String(value);
  }
}
```

### 2.6 筛选排序系统 (Filter & Sort)

#### 2.6.1 筛选条件处理

```typescript
// utils/filter.ts
export interface FilterCondition {
  fieldId: string;
  operator: FilterOperator;
  value?: any;
}

export enum FilterOperator {
  EQUALS = 'equals',
  NOT_EQUALS = 'notEquals',
  CONTAINS = 'contains',
  NOT_CONTAINS = 'notContains',
  STARTS_WITH = 'startsWith',
  ENDS_WITH = 'endsWith',
  IS_EMPTY = 'isEmpty',
  IS_NOT_EMPTY = 'isNotEmpty',
  GREATER_THAN = 'greaterThan',
  LESS_THAN = 'lessThan',
  GREATER_THAN_OR_EQUAL = 'greaterThanOrEqual',
  LESS_THAN_OR_EQUAL = 'lessThanOrEqual',
  IS_WITHIN = 'isWithin',
  IS_BEFORE = 'isBefore',
  IS_AFTER = 'isAfter',
  IS_ANY_OF = 'isAnyOf',
  IS_NONE_OF = 'isNoneOf',
}

export function applyFilters(
  records: Record[],
  filters: FilterCondition[],
  fields: Field[]
): Record[] {
  if (!filters.length) return records;

  return records.filter(record => {
    return filters.every(filter => {
      const field = fields.find(f => f.id === filter.fieldId);
      if (!field) return true;

      const value = record.values[filter.fieldId];
      return evaluateFilter(value, filter.operator, filter.value, field.type);
    });
  });
}

function evaluateFilter(
  value: CellValue,
  operator: FilterOperator,
  filterValue: any,
  fieldType: FieldType
): boolean {
  switch (operator) {
    case FilterOperator.EQUALS:
      return value === filterValue;
    
    case FilterOperator.NOT_EQUALS:
      return value !== filterValue;
    
    case FilterOperator.CONTAINS:
      return String(value).toLowerCase().includes(String(filterValue).toLowerCase());
    
    case FilterOperator.NOT_CONTAINS:
      return !String(value).toLowerCase().includes(String(filterValue).toLowerCase());
    
    case FilterOperator.STARTS_WITH:
      return String(value).toLowerCase().startsWith(String(filterValue).toLowerCase());
    
    case FilterOperator.ENDS_WITH:
      return String(value).toLowerCase().endsWith(String(filterValue).toLowerCase());
    
    case FilterOperator.IS_EMPTY:
      return value === null || value === undefined || value === '' || 
             (Array.isArray(value) && value.length === 0);
    
    case FilterOperator.IS_NOT_EMPTY:
      return value !== null && value !== undefined && value !== '' &&
             (!Array.isArray(value) || value.length > 0);
    
    case FilterOperator.GREATER_THAN:
      return Number(value) > Number(filterValue);
    
    case FilterOperator.LESS_THAN:
      return Number(value) < Number(filterValue);
    
    case FilterOperator.GREATER_THAN_OR_EQUAL:
      return Number(value) >= Number(filterValue);
    
    case FilterOperator.LESS_THAN_OR_EQUAL:
      return Number(value) <= Number(filterValue);
    
    case FilterOperator.IS_ANY_OF:
      if (Array.isArray(filterValue)) {
        return filterValue.includes(value);
      }
      return value === filterValue;
    
    case FilterOperator.IS_NONE_OF:
      if (Array.isArray(filterValue)) {
        return !filterValue.includes(value);
      }
      return value !== filterValue;
    
    default:
      return true;
  }
}
```

#### 2.7.1 Excel 处理

```typescript
// utils/export/excel.ts
import * as XLSX from 'xlsx';
import { saveAs } from 'file-saver';

export class ExcelExporter {
  static exportTable(
    tableName: string,
    fields: Field[],
    records: Record[]
  ): void {
    // 准备表头
    const headers = fields.map(f => f.name);
    
    // 准备数据
    const data = records.map(record => {
      return fields.map(field => {
        const value = record.values[field.id];
        return this.formatValueForExcel(value, field);
      });
    });

    // 创建工作簿
    const ws = XLSX.utils.aoa_to_sheet([headers, ...data]);
    const wb = XLSX.utils.book_new();
    XLSX.utils.book_append_sheet(wb, ws, tableName);

    // 导出文件
    const wbout = XLSX.write(wb, { bookType: 'xlsx', type: 'array' });
    const blob = new Blob([wbout], { type: 'application/octet-stream' });
    saveAs(blob, `${tableName}.xlsx`);
  }

  static async importTable(file: File): Promise<{
    headers: string[];
    data: any[][];
  }> {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload = (e) => {
        try {
          const data = new Uint8Array(e.target?.result as ArrayBuffer);
          const workbook = XLSX.read(data, { type: 'array' });
          const firstSheet = workbook.Sheets[workbook.SheetNames[0]];
          const jsonData = XLSX.utils.sheet_to_json(firstSheet, { header: 1 });
          
          resolve({
            headers: jsonData[0] as string[],
            data: jsonData.slice(1) as any[][]
          });
        } catch (error) {
          reject(error);
        }
      };
      reader.readAsArrayBuffer(file);
    });
  }

  private static formatValueForExcel(value: CellValue, field: Field): any {
    if (value === null || value === undefined) return '';
    
    switch (field.type) {
      case FieldType.MULTI_SELECT:
        return Array.isArray(value) ? value.join(', ') : value;
      
      case FieldType.DATE:
        return value;
      
      case FieldType.CHECKBOX:
        return value ? '是' : '否';
      
      default:
        return value;
    }
  }
}
```

### 3.1 大数据量处理

```typescript
// composables/useVirtualScroll.ts
import { ref, computed, onMounted, onUnmounted } from 'vue';

export function useVirtualScroll<T>(
  items: Ref<T[]>,
  options: {
    itemHeight: number;
    overscan?: number;
    containerHeight?: number;
  }
) {
  const { itemHeight, overscan = 5, containerHeight = 400 } = options;
  
  const scrollTop = ref(0);
  const containerRef = ref<HTMLElement | null>(null);

  // 计算可见范围
  const visibleRange = computed(() => {
    const startIdx = Math.floor(scrollTop.value / itemHeight);
    const visibleCount = Math.ceil(containerHeight / itemHeight);
    
    return {
      start: Math.max(0, startIdx - overscan),
      end: Math.min(items.value.length, startIdx + visibleCount + overscan)
    };
  });

  // 可见项目
  const visibleItems = computed(() => {
    return items.value.slice(visibleRange.value.start, visibleRange.value.end);
  });

  // 总高度
  const totalHeight = computed(() => items.value.length * itemHeight);

  // 偏移量
  const offsetY = computed(() => visibleRange.value.start * itemHeight);

  // 滚动处理
  const onScroll = (e: Event) => {
    scrollTop.value = (e.target as HTMLElement).scrollTop;
  };

  onMounted(() => {
    containerRef.value?.addEventListener('scroll', onScroll);
  });

  onUnmounted(() => {
    containerRef.value?.removeEventListener('scroll', onScroll);
  });

  return {
    containerRef,
    visibleItems,
    totalHeight,
    offsetY,
    startIndex: computed(() => visibleRange.value.start)
  };
}
```

### 3.2 计算缓存

```typescript
// composables/useComputedCache.ts
import { ref, computed, watch } from 'vue';

export function useComputedCache<T>(
  source: Ref<T[]>,
  computeKey: (item: T) => string,
  computeValue: (item: T) => any
) {
  const cache = ref(new Map<string, any>());
  const dirtyKeys = ref(new Set<string>());

  // 监听源数据变化
  watch(
    () => source.value,
    (newItems, oldItems) => {
      const oldKeys = new Set(oldItems?.map(computeKey) || []);
      const newKeys = new Set(newItems.map(computeKey));
      
      // 标记变更项
      newItems.forEach(item => {
        const key = computeKey(item);
        const oldItem = oldItems?.find(i => computeKey(i) === key);
        if (!oldItem || JSON.stringify(oldItem) !== JSON.stringify(item)) {
          dirtyKeys.value.add(key);
        }
      });
      
      // 清理已删除项
      oldKeys.forEach(key => {
        if (!newKeys.has(key)) {
          cache.value.delete(key);
        }
      });
    },
    { deep: true }
  );

  // 带缓存的计算
  const getComputed = (item: T): any => {
    const key = computeKey(item);
    
    if (dirtyKeys.value.has(key) || !cache.value.has(key)) {
      const value = computeValue(item);
      cache.value.set(key, value);
      dirtyKeys.value.delete(key);
      return value;
    }
    
    return cache.value.get(key);
  };

  return {
    getComputed,
    clearCache: () => {
      cache.value.clear();
      dirtyKeys.value.clear();
    }
  };
}
```

```typescript
// utils/debounce.ts
export function debounce<T extends (...args: any[]) => any>(
  fn: T,
  delay: number
): (...args: Parameters<T>) => void {
  let timer: ReturnType<typeof setTimeout> | null = null;
  
  return function (this: any, ...args: Parameters<T>) {
    if (timer) clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

export function throttle<T extends (...args: any[]) => any>(
  fn: T,
  limit: number
): (...args: Parameters<T>) => void {
  let inThrottle = false;
  
  return function (this: any, ...args: Parameters<T>) {
    if (!inThrottle) {
      fn.apply(this, args);
      inThrottle = true;
      setTimeout(() => inThrottle = false, limit);
    }
  };
}
```

```typescript
// utils/diff.ts
export function getObjectDiff(
  oldObj: Record<string, any>,
  newObj: Record<string, any>
): { added: string[]; removed: string[]; modified: string[] } {
  const oldKeys = Object.keys(oldObj);
  const newKeys = Object.keys(newObj);
  
  return {
    added: newKeys.filter(k => !oldKeys.includes(k)),
    removed: oldKeys.filter(k => !newKeys.includes(k)),
    modified: oldKeys.filter(k => 
      newKeys.includes(k) && oldObj[k] !== newObj[k]
    )
  };
}
```
