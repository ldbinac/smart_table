# SmartTable 技术架构设计文档

## 一、架构总览

### 1.1 架构设计理念

SmartTable 采用**纯前端分层架构**，所有数据持久化在浏览器本地，通过 IndexedDB 实现类数据库的存储能力。

```
┌─────────────────────────────────────────────────────────────────┐
│                         表现层 (Presentation)                    │
│  ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐   │
│  │ 表格视图 │ │ 看板视图 │ │ 日历视图 │ │ 甘特视图 │ │ 表单视图 │   │
│  └─────────┘ └─────────┘ └─────────┘ └─────────┘ └─────────┘   │
│  ┌─────────┐ ┌─────────┐                                        │
│  │ 画廊视图 │ │ 仪表盘  │                                        │
│  └─────────┘ └─────────┘                                        │
├─────────────────────────────────────────────────────────────────┤
│                         业务层 (Business)                        │
│  ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐   │
│  │ 数据管理 │ │ 视图管理 │ │ 公式引擎 │ │ 筛选排序 │ │ 导入导出 │   │
│  └─────────┘ └─────────┘ └─────────┘ └─────────┘ └─────────┘   │
│  ┌─────────┐ ┌─────────┐ ┌─────────┐                           │
│  │ 数据分组 │ │ 字段系统 │ │ 权限控制 │                           │
│  └─────────┘ └─────────┘ └─────────┘                           │
├─────────────────────────────────────────────────────────────────┤
│                         数据层 (Data)                            │
│  ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐ ┌─────────┐   │
│  │ Base存储 │ │ Table存储│ │ Field存储│ │Record存储│ │View存储 │   │
│  └─────────┘ └─────────┘ └─────────┘ └─────────┘ └─────────┘   │
├─────────────────────────────────────────────────────────────────┤
│                         存储层 (Storage)                         │
│              ┌─────────────────────────────┐                   │
│              │       IndexedDB API         │                   │
│              └─────────────────────────────┘                   │
└─────────────────────────────────────────────────────────────────┘
```

## 二、核心模块详细设计

### 2.1 数据存储层 (Storage Layer)

#### 2.1.1 IndexedDB Schema 设计

```typescript
// db/schema.ts
import Dexie, { Table } from 'dexie';

// 多维表格
export interface Base {
  id: string;
  name: string;
  description?: string;
  icon?: string;
  color?: string;
  isStarred: boolean;
  createdAt: number;
  updatedAt: number;
}

// 数据表
export interface Table {
  id: string;
  baseId: string;
  name: string;
  description?: string;
  primaryFieldId: string;
  recordCount: number;
  order: number;
  createdAt: number;
  updatedAt: number;
}

// 字段定义
export interface Field {
  id: string;
  tableId: string;
  name: string;
  type: FieldType;  // 22种字段类型
  options?: FieldOptions;
  isPrimary: boolean;
  isSystem: boolean;
  isRequired: boolean;
  defaultValue?: any;
  description?: string;
  order: number;
  createdAt: number;
  updatedAt: number;
}

// 数据记录
export interface Record {
  id: string;
  tableId: string;
  values: Record<string, CellValue>;
  createdAt: number;
  updatedAt: number;
  createdBy?: string;
  updatedBy?: string;
}

// 视图配置
export interface View {
  id: string;
  tableId: string;
  name: string;
  type: ViewType;  // 6种视图类型
  config: ViewConfig;
  filters: FilterCondition[];
  sorts: SortConfig[];
  groupBys: string[];
  hiddenFields: string[];
  frozenFields: string[];
  rowHeight: 'short' | 'medium' | 'tall';
  isDefault: boolean;
  order: number;
  createdAt: number;
  updatedAt: number;
}

// 仪表盘
export interface Dashboard {
  id: string;
  baseId: string;
  name: string;
  description?: string;
  widgets: WidgetConfig[];
  layout: LayoutConfig;
  createdAt: number;
  updatedAt: number;
}

// 附件存储
export interface Attachment {
  id: string;
  recordId: string;
  fieldId: string;
  name: string;
  size: number;
  type: string;
  data: Blob;
  thumbnail?: Blob;
  createdAt: number;
}

// 操作历史
export interface OperationHistory {
  id: string;
  baseId: string;
  tableId?: string;
  recordId?: string;
  fieldId?: string;
  action: 'create' | 'update' | 'delete';
  entityType: 'base' | 'table' | 'field' | 'record' | 'view';
  oldValue?: any;
  newValue?: any;
  timestamp: number;
  userId?: string;
}
```

#### 2.5.2 支持的函数分类

| 类别 | 函数数量 | 函数列表 |

#### 2.5.3 公式计算流程

```
公式字符串 → 解析字段引用 → 计算函数 → 计算表达式 → 返回结果
```
